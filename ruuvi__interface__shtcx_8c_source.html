76static rd_status_t SHTCX_TO_RUUVI_ERROR (const int16_t rslt)
77{
78 if (STATUS_OK == rslt) { return RD_SUCCESS; }
79
80 rd_status_t err_code = RD_ERROR_INTERNAL;
81
82 if (STATUS_UNKNOWN_DEVICE == rslt) { err_code = RD_ERROR_NOT_FOUND; }
83 else if (STATUS_ERR_BAD_DATA == rslt) { err_code = RD_ERROR_INVALID_DATA; }
84 else if (STATUS_CRC_FAIL == rslt) { err_code = RD_ERROR_INVALID_DATA; }
85 else if (STATUS_WAKEUP_FAILED == rslt) { err_code = RD_ERROR_INTERNAL; }
86 else if (STATUS_SLEEP_FAILED == rslt) { err_code = RD_ERROR_INTERNAL; }
87
88 return err_code;
89}

91rd_status_t ri_shtcx_init (rd_sensor_t * sensor, rd_bus_t bus, uint8_t handle)
92{
93 rd_status_t err_code = RD_SUCCESS;
94
95 if (NULL == sensor)
96 {
97 err_code |= RD_ERROR_NULL;
98 }
99 else if (rd_sensor_is_init (sensor))
100 {
101 err_code |= RD_ERROR_INVALID_STATE;
102 }
103 else
104 {
105 rd_sensor_initialize (sensor);
106 sensor->name = m_sensor_name;
107 uint8_t retries = 0;
108
109 switch (bus)
110 {
111 case RD_BUS_I2C:
112 do
113 {
114 err_code = SHTCX_TO_RUUVI_ERROR (shtc1_probe());
115 retries++;
116 } while ( (RD_ERROR_INVALID_DATA == err_code)
117 && (retries < SHTCX_PROBE_RETRIES_MAX));
118
119 break;
120
121 default:
122 err_code |= RD_ERROR_INVALID_PARAM;
123 }
124
125 if (RD_SUCCESS != err_code)
126 {
127 err_code = RD_ERROR_NOT_FOUND;
128 }
129 else
130 {
131 // Sensirion driver delays high-power mode time in any case.
132 // Explicitly entering low-power mode has no effect.
133 shtc1_enable_low_power_mode (0);
134 sensor->init = ri_shtcx_init;
135 sensor->uninit = ri_shtcx_uninit;
136 sensor->samplerate_set = ri_shtcx_samplerate_set;
137 sensor->samplerate_get = ri_shtcx_samplerate_get;
138 sensor->resolution_set = ri_shtcx_resolution_set;
139 sensor->resolution_get = ri_shtcx_resolution_get;
140 sensor->scale_set = ri_shtcx_scale_set;
141 sensor->scale_get = ri_shtcx_scale_get;
142 sensor->dsp_set = ri_shtcx_dsp_set;
143 sensor->dsp_get = ri_shtcx_dsp_get;
144 sensor->mode_set = ri_shtcx_mode_set;
145 sensor->mode_get = ri_shtcx_mode_get;
146 sensor->data_get = ri_shtcx_data_get;
147 sensor->configuration_set = rd_sensor_configuration_set;
148 sensor->configuration_get = rd_sensor_configuration_get;
149 sensor->provides.datas.temperature_c = 1;
150 sensor->provides.datas.humidity_rh = 1;
151 err_code |= SHTCX_TO_RUUVI_ERROR (shtc1_sleep());
152 m_tsample = RD_UINT64_INVALID;
153 m_is_init = true;
154 }
155 }
156
157 return err_code;
158}

267// Start single on command, mark autorefresh with continuous
268rd_status_t ri_shtcx_mode_set (uint8_t * mode)
269{
270 rd_status_t err_code = RD_SUCCESS;
271
272 if (NULL == mode)
273 {
274 err_code |= RD_ERROR_NULL;
275 }
276 // Enter sleep by default and by explicit sleep commmand
277 else if ( (RD_SENSOR_CFG_SLEEP == *mode) || (RD_SENSOR_CFG_DEFAULT == *mode))
278 {
279 m_autorefresh = false;
280 *mode = RD_SENSOR_CFG_SLEEP;
281 err_code |= SHTCX_TO_RUUVI_ERROR (shtc1_sleep());
282 }
283 else if (RD_SENSOR_CFG_SINGLE == *mode)
284 {
285 // Do nothing if sensor is in continuous mode
286 uint8_t current_mode;
287 ri_shtcx_mode_get (&current_mode);
288
289 if (RD_SENSOR_CFG_CONTINUOUS == current_mode)
290 {
291 *mode = RD_SENSOR_CFG_CONTINUOUS;
292 return RD_ERROR_INVALID_STATE;
293 }
294
295 // Enter sleep after measurement
296 m_autorefresh = false;
297 *mode = RD_SENSOR_CFG_SLEEP;
298 m_tsample = rd_sensor_timestamp_get();
299 err_code |= SHTCX_TO_RUUVI_ERROR (shtc1_wake_up());
300 sensirion_sleep_usec (RI_SHTCX_WAKEUP_US);
301 err_code |= SHTCX_TO_RUUVI_ERROR (shtc1_measure_blocking_read (&m_temperature,
302 &m_humidity));
303 err_code |= SHTCX_TO_RUUVI_ERROR (shtc1_sleep());
304 return err_code;
305 }
306 else if (RD_SENSOR_CFG_CONTINUOUS == *mode)
307 {
308 m_autorefresh = true;
309 err_code |= RD_SUCCESS;
310 }
311 else
312 {
313 err_code |= RD_ERROR_INVALID_PARAM;
314 }
315
316 return err_code;
317}

336rd_status_t ri_shtcx_data_get (rd_sensor_data_t * const p_data)
337{
338 rd_status_t err_code = RD_SUCCESS;
339
340 if (NULL == p_data)
341 {
342 err_code |= RD_ERROR_NULL;
343 }
344 else
345 {
346 if (m_autorefresh)
347 {
348 // Set autorefresh to false to take a single sample
349 m_autorefresh = false;
350 uint8_t mode = RD_SENSOR_CFG_SINGLE;
351 err_code |= ri_shtcx_mode_set (&mode);
352 // Restore autorefresh
353 m_autorefresh = true;
354 }
355
356 if ( (RD_SUCCESS == err_code) && (RD_UINT64_INVALID != m_tsample))
357 {
358 rd_sensor_data_t d_environmental;
359 rd_sensor_data_fields_t env_fields = {.bitfield = 0};
360 float env_values[2];
361 env_values[0] = m_humidity / 1000.0f;
362 env_values[1] = m_temperature / 1000.0f;
363 env_fields.datas.humidity_rh = 1;
364 env_fields.datas.temperature_c = 1;
365 d_environmental.data = env_values;
366 d_environmental.valid = env_fields;
367 d_environmental.fields = env_fields;
368 d_environmental.timestamp_ms = m_tsample;
369 rd_sensor_data_populate (p_data,
370 &d_environmental,
371 p_data->fields);
372 }
373 }
374
375 return err_code;
376}
