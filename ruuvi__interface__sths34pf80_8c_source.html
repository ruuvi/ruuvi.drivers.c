81static rd_status_t st_to_ruuvi_error (const int32_t st_err)
82{
83 if (0 == st_err)
84 {
85 return RD_SUCCESS;
86 }
87
88 // Negative values are ST driver internal errors
89 if (st_err < 0)
90 {
91 return RD_ERROR_INTERNAL;
92 }
93
94 // Positive values are platform errors passed through from I2C adapter.
95 // These are already Ruuvi error codes (RD_ERROR_*), so return directly.
96 // Common cases:
97 // - RD_ERROR_NOT_ACKNOWLEDGED: Device not responding (wrong address or not present)
98 // - RD_ERROR_TIMEOUT: I2C bus timeout
99 // - RD_ERROR_BUSY: I2C bus busy
100 // - RD_ERROR_NULL: NULL pointer passed to I2C function
101 return (rd_status_t) st_err;
102}

112static rd_status_t read_sample (void)
113{
114 rd_status_t err_code = RD_SUCCESS;
115 int32_t st_err = 0;
116 sths34pf80_func_status_t func_status = {0};
117 // Read FUNC_STATUS to get presence/motion flags
118 st_err = sths34pf80_func_status_get (&m_ctx.ctx, &func_status);
119 err_code |= st_to_ruuvi_error (st_err);
120 // Read ambient and object temperatures
121 st_err = sths34pf80_tambient_raw_get (&m_ctx.ctx, &m_ctx.tambient_raw);
122 err_code |= st_to_ruuvi_error (st_err);
123 st_err = sths34pf80_tobject_raw_get (&m_ctx.ctx, &m_ctx.tobject_raw);
124 err_code |= st_to_ruuvi_error (st_err);
125 // Store flags
126 m_ctx.presence_flag = func_status.pres_flag;
127 m_ctx.motion_flag = func_status.mot_flag;
128 // Record timestamp
129 m_ctx.tsample = rd_sensor_timestamp_get();
130 return err_code;

158rd_status_t ri_sths34pf80_init (rd_sensor_t * p_sensor, rd_bus_t bus, uint8_t handle)
159{
160 rd_status_t err_code = RD_SUCCESS;
161 int32_t st_err = 0;
162
163 if (NULL == p_sensor)
164 {
165 return RD_ERROR_NULL;
166 }
167
168 if (m_is_init)
169 {
170 return RD_ERROR_INVALID_STATE;
171 }
172
173 // Only I2C is supported
174 if (RD_BUS_I2C != bus)
175 {
176 return RD_ERROR_INVALID_PARAM;
177 }
178
179 rd_sensor_initialize (p_sensor);
180 p_sensor->name = m_sensor_name;
181 // Setup I2C context
182 m_ctx.handle = handle;
183 m_ctx.ctx.handle = &m_ctx.handle;
184 m_ctx.ctx.write_reg = ri_i2c_sths34pf80_write;
185 m_ctx.ctx.read_reg = ri_i2c_sths34pf80_read;
186 m_ctx.ctx.mdelay = sths34pf80_mdelay;
187 // Check WHO_AM_I
188 uint8_t whoami = 0;
189 st_err = sths34pf80_device_id_get (&m_ctx.ctx, &whoami);
190 err_code |= st_to_ruuvi_error (st_err);
191
192 // If read failed, return the communication error
193 if (RD_SUCCESS != err_code)
194 {
195 rd_sensor_uninitialize (p_sensor);
196 return err_code;
197 }
198
199 // If read succeeded but ID doesn't match, sensor is wrong type
200 if (STHS34PF80_ID != whoami)
201 {
202 rd_sensor_uninitialize (p_sensor);
203 return RD_ERROR_NOT_FOUND;
204 }
205
206 // Reset device
207 st_err = sths34pf80_boot_set (&m_ctx.ctx, 1);
208 err_code |= st_to_ruuvi_error (st_err);
209 ri_delay_ms (BOOT_DELAY_MS);
210 // Configure default settings: ODR off (sleep mode), BDU enabled
211 st_err = sths34pf80_odr_set (&m_ctx.ctx, STHS34PF80_ODR_OFF);
212 err_code |= st_to_ruuvi_error (st_err);
213 st_err = sths34pf80_block_data_update_set (&m_ctx.ctx, 1);
214 err_code |= st_to_ruuvi_error (st_err);
215
216 if (RD_SUCCESS != err_code)
217 {
218 rd_sensor_uninitialize (p_sensor);
219 return err_code;
220 }
221
222 // Initialize context
223 m_ctx.odr = STHS34PF80_ODR_OFF;
224 m_ctx.mode = RD_SENSOR_CFG_SLEEP;
225 m_ctx.autorefresh = false;
226 m_ctx.tsample = RD_UINT64_INVALID;
227 m_ctx.tambient_raw = 0;
228 m_ctx.tobject_raw = 0;
229 m_ctx.presence_flag = 0;
230 m_ctx.motion_flag = 0;
231 // Setup sensor struct
232 p_sensor->init = ri_sths34pf80_init;
233 p_sensor->uninit = ri_sths34pf80_uninit;
234 p_sensor->samplerate_set = ri_sths34pf80_samplerate_set;
235 p_sensor->samplerate_get = ri_sths34pf80_samplerate_get;
236 p_sensor->resolution_set = ri_sths34pf80_resolution_set;
237 p_sensor->resolution_get = ri_sths34pf80_resolution_get;
238 p_sensor->scale_set = ri_sths34pf80_scale_set;
239 p_sensor->scale_get = ri_sths34pf80_scale_get;
240 p_sensor->dsp_set = ri_sths34pf80_dsp_set;
241 p_sensor->dsp_get = ri_sths34pf80_dsp_get;
242 p_sensor->mode_set = ri_sths34pf80_mode_set;
243 p_sensor->mode_get = ri_sths34pf80_mode_get;
244 p_sensor->data_get = ri_sths34pf80_data_get;
245 p_sensor->configuration_set = rd_sensor_configuration_set;
246 p_sensor->configuration_get = rd_sensor_configuration_get;
247 // Define provided data fields
248 p_sensor->provides.datas.temperature_c = 1;
249 p_sensor->provides.datas.presence = 1;
250 p_sensor->provides.datas.motion = 1;
251 p_sensor->provides.datas.ir_object = 1;
252 m_is_init = true;
253 return err_code;
254}

291static rd_status_t get_max_odr_for_oversampling (sths34pf80_odr_t * const p_max_odr,
292 uint8_t * const p_max_rate)
293{
294 rd_status_t err_code = RD_SUCCESS;
295 sths34pf80_avg_tobject_num_t avg_tmos;
296 int32_t st_err = sths34pf80_avg_tobject_num_get (&m_ctx.ctx, &avg_tmos);
297 err_code = st_to_ruuvi_error (st_err);
298
299 if (RD_SUCCESS != err_code)
300 {
301 return err_code;
302 }
303
304 switch (avg_tmos)
305 {
306 case STHS34PF80_AVG_TMOS_2:
307 case STHS34PF80_AVG_TMOS_8:
308 case STHS34PF80_AVG_TMOS_32:
309 default:
310 *p_max_odr = STHS34PF80_ODR_AT_30Hz;
311 *p_max_rate = 30U;
312 break;
313
314 case STHS34PF80_AVG_TMOS_128:
315 *p_max_odr = STHS34PF80_ODR_AT_8Hz;
316 *p_max_rate = 8U;
317 break;
318
319 case STHS34PF80_AVG_TMOS_256:
320 *p_max_odr = STHS34PF80_ODR_AT_4Hz;
321 *p_max_rate = 4U;
322 break;
323
324 case STHS34PF80_AVG_TMOS_512:
325 *p_max_odr = STHS34PF80_ODR_AT_2Hz;
326 *p_max_rate = 2U;
327 break;
328
329 case STHS34PF80_AVG_TMOS_1024:
330 *p_max_odr = STHS34PF80_ODR_AT_1Hz;
331 *p_max_rate = 1U;
332 break;
333
334 case STHS34PF80_AVG_TMOS_2048:
335 *p_max_odr = STHS34PF80_ODR_AT_0Hz50;
336 *p_max_rate = RI_STHS34PF80_SAMPLERATE_0HZ50;
337 break;
338 }
339
340 return RD_SUCCESS;
341}

343rd_status_t ri_sths34pf80_samplerate_set (uint8_t * samplerate)
344{
345 rd_status_t err_code = RD_SUCCESS;
346
347 if (NULL == samplerate)
348 {
349 return RD_ERROR_NULL;
350 }
351
352 VERIFY_SENSOR_SLEEPS();
353 sths34pf80_odr_t odr = STHS34PF80_ODR_AT_1Hz;
354
355 if (RD_SENSOR_CFG_NO_CHANGE == *samplerate)
356 {
357 return ri_sths34pf80_samplerate_get (samplerate);
358 }
359 else if (RI_STHS34PF80_SAMPLERATE_0HZ25 == *samplerate ||
360 RD_SENSOR_CFG_MIN == *samplerate)
361 {
362 // 0.25 Hz - lowest power, 4 second period
363 odr = STHS34PF80_ODR_AT_0Hz25;
364 *samplerate = RI_STHS34PF80_SAMPLERATE_0HZ25;
365 }
366 else if (RI_STHS34PF80_SAMPLERATE_0HZ50 == *samplerate)
367 {
368 // 0.5 Hz - 2 second period
369 odr = STHS34PF80_ODR_AT_0Hz50;
370 *samplerate = RI_STHS34PF80_SAMPLERATE_0HZ50;
371 }
372 else if (RD_SENSOR_CFG_DEFAULT == *samplerate ||
373 1U == *samplerate)
374 {
375 odr = STHS34PF80_ODR_AT_1Hz;
376 *samplerate = 1;
377 }
378 else if (RD_SENSOR_CFG_MAX == *samplerate)
379 {
380 // Get maximum ODR based on current oversampling setting
381 sths34pf80_odr_t max_odr;
382 uint8_t max_rate;
383 err_code = get_max_odr_for_oversampling (&max_odr, &max_rate);
384
385 if (RD_SUCCESS != err_code)
386 {
387 return err_code;
388 }
389
390 odr = max_odr;
391 *samplerate = max_rate;
392 }
393 else if (*samplerate <= 1U)
394 {
395 odr = STHS34PF80_ODR_AT_1Hz;
396 *samplerate = 1;
397 }
398 else if (*samplerate <= 2U)
399 {
400 odr = STHS34PF80_ODR_AT_2Hz;
401 *samplerate = 2;
402 }
403 else if (*samplerate <= 4U)
404 {
405 odr = STHS34PF80_ODR_AT_4Hz;
406 *samplerate = 4;
407 }
408 else if (*samplerate <= 8U)
409 {
410 odr = STHS34PF80_ODR_AT_8Hz;
411 *samplerate = 8;
412 }
413 else if (*samplerate <= 15U)
414 {
415 odr = STHS34PF80_ODR_AT_15Hz;
416 *samplerate = 15;
417 }
418 else if (*samplerate <= 30U)
419 {
420 odr = STHS34PF80_ODR_AT_30Hz;
421 *samplerate = 30;
422 }
423 else
424 {
425 *samplerate = RD_SENSOR_ERR_NOT_SUPPORTED;
426 err_code |= RD_ERROR_NOT_SUPPORTED;
427 }
428
429 // Check if requested ODR exceeds max allowed by current oversampling setting
430 if (RD_SUCCESS == err_code)
431 {
432 sths34pf80_odr_t max_odr;
433 uint8_t max_rate;
434 rd_status_t max_err = get_max_odr_for_oversampling (&max_odr, &max_rate);
435
436 if (RD_SUCCESS == max_err)
437 {
438 if (odr > max_odr)
439 {
440 // Cannot achieve requested rate with current oversampling
441 *samplerate = RD_SENSOR_ERR_NOT_SUPPORTED;
442 err_code |= RD_ERROR_NOT_SUPPORTED;
443 }
444 }
445 else
446 {
447 err_code |= max_err;
448 }
449 }
450
451 // Note: ODR is only applied when mode is set to continuous
452 if (RD_SUCCESS == err_code)
453 {
454 m_ctx.odr = odr;
455 }
456
457 return err_code;
458}

598rd_status_t ri_sths34pf80_mode_set (uint8_t * mode)
599{
600 rd_status_t err_code = RD_SUCCESS;
601 int32_t st_err = 0;
602
603 if (NULL == mode)
604 {
605 return RD_ERROR_NULL;
606 }
607
608 if (RD_SENSOR_CFG_SLEEP == *mode || RD_SENSOR_CFG_DEFAULT == *mode)
609 {
610 st_err = sths34pf80_odr_set (&m_ctx.ctx, STHS34PF80_ODR_OFF);
611 err_code |= st_to_ruuvi_error (st_err);
612 m_ctx.mode = RD_SENSOR_CFG_SLEEP;
613 m_ctx.autorefresh = false;
614 *mode = RD_SENSOR_CFG_SLEEP;
615 }
616 else if (RD_SENSOR_CFG_SINGLE == *mode)
617 {
618 // If in continuous mode, cannot do single
619 if (RD_SENSOR_CFG_CONTINUOUS == m_ctx.mode)
620 {
621 *mode = RD_SENSOR_CFG_CONTINUOUS;
622 return RD_ERROR_INVALID_STATE;
623 }
624
625 // Trigger one-shot measurement using max sample rate for fastest response
626 uint8_t samplerate = RD_SENSOR_CFG_MAX;
627 err_code = ri_sths34pf80_samplerate_set (&samplerate);
628
629 if (RD_SUCCESS == err_code)
630 {
631 // Apply ODR to hardware
632 st_err = sths34pf80_odr_set (&m_ctx.ctx, m_ctx.odr);
633 err_code |= st_to_ruuvi_error (st_err);
634 }
635
636 if (RD_SUCCESS == err_code)
637 {
638 // Wait for sample based on configured rate (add margin)
639 uint32_t delay_ms = (1000U / samplerate) + 10U;
640 ri_delay_ms (delay_ms);
641 // Read data
642 err_code |= read_sample();
643 }
644
645 // Return to sleep
646 st_err = sths34pf80_odr_set (&m_ctx.ctx, STHS34PF80_ODR_OFF);
647 err_code |= st_to_ruuvi_error (st_err);
648 m_ctx.mode = RD_SENSOR_CFG_SLEEP;
649 *mode = RD_SENSOR_CFG_SLEEP;
650 }
651 else if (RD_SENSOR_CFG_CONTINUOUS == *mode)
652 {
653 // Use default samplerate if ODR was never configured
654 if (STHS34PF80_ODR_OFF == m_ctx.odr)
655 {
656 uint8_t samplerate = RD_SENSOR_CFG_DEFAULT;
657 err_code = ri_sths34pf80_samplerate_set (&samplerate);
658 }
659
660 if (RD_SUCCESS == err_code)
661 {
662 st_err = sths34pf80_odr_set (&m_ctx.ctx, m_ctx.odr);
663 err_code |= st_to_ruuvi_error (st_err);
664 }
665
666 if (RD_SUCCESS == err_code)
667 {
668 m_ctx.mode = RD_SENSOR_CFG_CONTINUOUS;
669 m_ctx.autorefresh = true;
670 *mode = RD_SENSOR_CFG_CONTINUOUS;
671 }
672 }
673 else
674 {
675 err_code |= RD_ERROR_INVALID_PARAM;
676 }
677
678 return err_code;
679}

692rd_status_t ri_sths34pf80_data_get (rd_sensor_data_t * const data)
693{
694 rd_status_t err_code = RD_SUCCESS;
695
696 if (NULL == data)
697 {
698 return RD_ERROR_NULL;
699 }
700
701 // If in continuous mode, check DRDY and read only if new data is available
702 if (m_ctx.autorefresh)
703 {
704 sths34pf80_drdy_status_t drdy_status = {0};
705 int32_t st_err = sths34pf80_drdy_status_get (&m_ctx.ctx, &drdy_status);
706 err_code |= st_to_ruuvi_error (st_err);
707
708 if ( (RD_SUCCESS == err_code) && (drdy_status.drdy))
709 {
710 err_code |= read_sample();
711 }
712
713 // If no new data, we return the cached values with their original timestamp
714 }
715
716 // Populate data structure
717 // Note: Only fields that are requested AND provided will be populated
718 rd_sensor_data_t d_environmental = {0};
719 rd_sensor_data_fields_t env_fields = {0};
720 // Ambient temperature in Celsius
721 float values[4] = {0};
722 values[0] = tambient_to_celsius (m_ctx.tambient_raw);
723 values[1] = (float) m_ctx.presence_flag;
724 values[2] = (float) m_ctx.motion_flag;
725 values[3] = (float) m_ctx.tobject_raw; // Dimensionless IR signal
726
727 // Only mark fields valid if a sample has actually been taken
728 if (RD_UINT64_INVALID != m_ctx.tsample)
729 {
730 env_fields.datas.temperature_c = 1;
731 env_fields.datas.presence = 1;
732 env_fields.datas.motion = 1;
733 env_fields.datas.ir_object = 1;
734 }
735
736 d_environmental.data = values;
737 d_environmental.valid = env_fields;
738 d_environmental.fields = env_fields;
739 d_environmental.timestamp_ms = m_ctx.tsample;
740 rd_sensor_data_populate (data, &d_environmental, data->fields);
741 data->timestamp_ms = m_ctx.tsample;
742 return err_code;
743}
