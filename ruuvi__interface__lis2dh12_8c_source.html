1149  cfg.ylie = PROPERTY_ENABLE;
1150  cfg.zlie = PROPERTY_ENABLE;
1151  */

1153  // 1 LSb = 16 mg @ FS = 2 g
1154  // 1 LSb = 32 mg @ FS = 4 g
1155  // 1 LSb = 62 mg @ FS = 8 g
